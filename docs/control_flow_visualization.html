5. 关键代码解析

// pto_orchestrator.cpp: pto2_submit_task() void pto2_submit_task(...) { // Step 1: 分配任务槽位 int32_t task_id = task_ring.pto2_task_ring_alloc(); // Step 2: 查找依赖 (通过 TensorMap) for (auto& param : params) { if (param.type == INPUT) { // 查找生产者 tensor_map.lookup(param.tensor, lookup_result); // 构建依赖链 for (auto& entry : lookup_result) { pto2_add_consumer_to_producer(producer, task_id); } } } // Step 3: 分配输出缓冲区 buffer = heap_ring.pto2_heap_ring_alloc(total_size); // Step 4: 初始化调度器状态 scheduler.init_task(task_id, task); }
// pto_scheduler.cpp: on_task_complete() void on_task_complete(task_id) { // Step 1: 标记完成 task_state[slot] = COMPLETED; // Step 2: 通知消费者 (遍历 fanout) for (consumer in task.fanout_list) { // 增加消费者的 fanin_refcount fanin_refcount[consumer]++; // 检查是否所有依赖都满足了 if (fanin_refcount[consumer] == fanin_count[consumer]) { // 变为 READY，放入就绪队列 task_state[consumer] = READY; ready_queue.push(consumer); } } // Step 3: 释放生产者 (遍历 fanin) for (producer in task.fanin_list) { fanout_refcount[producer]++; check_and_handle_consumed(producer); } }
// pto_scheduler.h: release_fanin_and_check_ready() void release_fanin_and_check_ready(task_id, task) { // 原子操作: fanin_refcount + 1 int32_t new_refcount = __atomic_fetch_add(&fanin_refcount[slot], 1) + 1; // 检查是否所有生产者都完成了 if (new_refcount == task->fanin_count) { // CAS 操作: PENDING -> READY // (防止多个线程同时将同一任务入队) if (CAS(&task_state[slot], PENDING, READY)) { ready_queue.push(task_id); } } }

6. 设计精髓

🔄 零拷贝

Ring Buffer 避免内存复制，任务描述符在共享内存中，多处理器直接访问

🔍 自动依赖发现

TensorMap 通过内存地址自动发现依赖，用户只需指定输入输出

⚡ 背压控制

环形缓冲区满时自旋等待，防止内存溢出，实现自动流控

🔓 无锁并发

原子操作 (CAS, fetch_add) 保证线程安全，避免锁的开销

♻️ 隐式回收

不需要显式 free()，last_task_alive 推进时自动回收

📦 作用域生命周期

scope_begin/end 管理任务组生命周期，作用域结束时自动释放引用
